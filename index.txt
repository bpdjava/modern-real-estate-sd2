        // Function to show custom alert modal
        function showAlert(title, message) {

                });
                // Close mobile menu if open
                const mobileMenu = document.getElementById('mobile-menu');
                if (!mobileMenu.classList.contains('hidden')) {
                    mobileMenu.classList.add('hidden');
                }
            });
        });

        // Mobile menu toggle
        document.getElementById('mobile-menu-button').addEventListener('click', function() {
            const mobileMenu = document.getElementById('mobile-menu');
            mobileMenu.classList.toggle('hidden');
        });

        // Header scroll effect
        window.addEventListener('scroll', function() {
            const header = document.querySelector('header');
            if (window.scrollY > 0) {
                header.classList.add('scrolled');
            } else {
                header.classList.remove('scrolled');
            }
        });


        // Commission Savings Calculator Logic
        document.getElementById('calculateSavings').addEventListener('click', function() {
            const homeValueInput = document.getElementById('homeValue');
            const homeValue = parseFloat(homeValueInput.value);

            // Clear previous results if input is invalid
            if (isNaN(homeValue) || homeValue <= 0) {
                document.getElementById('traditionalCommission').textContent = '$0';
                document.getElementById('modernRealEstateFee').textContent = '$0';
                document.getElementById('potentialSavings').textContent = '$0';
                showAlert('Invalid Input', 'Please enter a valid home value (a positive number).');
                return;
            }

            const traditionalCommissionRate = 0.05; // Assuming a typical 5% commission
            const traditionalCommission = homeValue * traditionalCommissionRate;

            let modernRealEstateFee = 0;
            if (homeValue < 500000) {
                modernRealEstateFee = 5000;
            } else if (homeValue >= 500000 && homeValue < 1500000) {
                modernRealEstateFee = 10000;
            } else { // homeValue >= 1500000
                modernRealEstateFee = 15000;
            }

            const potentialSavings = traditionalCommission - modernRealEstateFee;

            // Update the display with formatted numbers
            document.getElementById('traditionalCommission').textContent = `$${traditionalCommission.toLocaleString('en-US')}`;
            document.getElementById('modernRealEstateFee').textContent = `$${modernRealEstateFee.toLocaleString('en-US')}`;
            document.getElementById('potentialSavings').textContent = `$${potentialSavings.toLocaleString('en-US')}`;
            document.getElementById('commissionRate').textContent = `${(traditionalCommissionRate * 100)}%`;
            document.getElementById('disclaimerRate').textContent = `${(traditionalCommissionRate * 100)}%`;
        });

        // Function to generate and set the AI image
        async function generateAndSetImage() {
            const heroBackground = document.getElementById('hero-background');
            const loader = document.getElementById('hero-image-loader');

            // Show loader
            loader.classList.remove('hidden');
            heroBackground.style.opacity = '0.1'; // Dim background during loading

            // Updated prompt for a San Diego modern beach house
            const prompt = "Modern San Diego beach house, contemporary architecture, large windows, ocean view, palm trees, sunny day, high-quality, professional photography style.";
            const payload = { instances: { prompt: prompt }, parameters: { "sampleCount": 1} };
            const apiKey = ""; // Canvas will provide this at runtime
            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/imagen-3.0-generate-002:predict?key=${apiKey}`;

            try {
                const response = await fetch(apiUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });
                const result = await response.json();

                if (result.predictions && result.predictions.length > 0 && result.predictions[0].bytesBase64Encoded) {
                    const imageUrl = `data:image/png;base64,${result.predictions[0].bytesBase64Encoded}`;
                    heroBackground.style.backgroundImage = `url('${imageUrl}')`;
                    heroBackground.style.opacity = '1'; // Restore full opacity after loading
                } else {
                    console.error('Failed to generate image: Invalid response structure.');
                    heroBackground.style.backgroundImage = `url('https://placehold.co/1920x1080/000000/FFFFFF?text=Image+Load+Error')`;
                    heroBackground.style.opacity = '1';
                }
            } catch (error) {
                console.error('Error generating image:', error);
                heroBackground.style.backgroundImage = `url('https://placehold.co/1920x1080/000000/FFFFFF?text=Image+Load+Error')`;
                heroBackground.style.opacity = '1';
            } finally {
                // Hide loader regardless of success or failure
                loader.classList.add('hidden');
            }
        }

        // Call the image generation function when the window loads
        window.onload = generateAndSetImage;
    </script>
